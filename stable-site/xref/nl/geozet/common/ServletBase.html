111     /**
112      * Maakt de parameter string, onafhankelijk van de toegangsmethode (POST of
113      * GET).
114      * 
115      * @param request
116      *            het request van dit service verzoek
117      * @param excludeparam
118      *            een request parameter die eventueel uitgesloten dient te
119      *            worden
120      * @return de string begint met een ampersand (&amp;)
121      */
122     protected final String buildQueryString(HttpServletRequest request,
123             String excludeparam) {
124         /* want servlet api is niet generic */
125         @SuppressWarnings("unchecked")
126         final Map<String, String[]> reqs = request.getParameterMap();
127         final StringBuilder s = new StringBuilder();
128         String key;
129         String[] vals;
130 
131         for (final Map.Entry<String, String[]> entry : reqs.entrySet()) {
132             key = entry.getKey();
133             if (key.equals(excludeparam)) {
134                 continue;
135             }
136             s.append("&amp;").append(key).append("=");
137             vals = entry.getValue();
138             final int count = vals.length;
139             if (count > 1) {
140                 for (int i = 0; i < count; i++) {
141                     s.append(vals[i]);
142                     s.append("&amp;").append(key).append("=");
143                 }
144                 s.delete(s.lastIndexOf("&amp;"), s.length());
145             } else {
146                 s.append(vals[0]);
147             }
148         }
149         // dit moet soms wel soms niet..dus maar even niet: s.delete(0,
150         // "&amp;".length());
151         return s.toString();
152     }

154     /**
155      * Geeft een paginering in HTML formaat. Als in een resultaatlijst
156      * (bijvoorbeeld bij de zoekpagina) meer dan een configureerbaar aantal
157      * resultaten gevonden worden, komen er meerdere pagina's met
158      * zoekresultaten. In dit geval komt er boven en onder de resultaatlijst een
159      * paginanummering te staan. Deze bestaat uit een aantal elementen:
160      * <dl>
161      * <dt>Vorige</dt>
162      * <dd>Indien de huidige resultaatpagina niet de eerste is begint de
163      * paginanummering met een link 'Vorige'</dd>
164      * <dt>Volgende</dt>
165      * <dd>Indien de huidige resultaatpagina niet de laatste is eindigt de
166      * paginanummering met een link 'Volgende'</dd>
167      * <dt>Paginanummers</dt>
168      * <dd>Een maximum van 7 paginanummers staat in de lijst. Dit is de huidige
169      * pagina, twee ervoor, twee erna en de eerste en laatste pagina. De
170      * volgende regels gelden hiervoor:
171      * <ul>
172      * <li>Indien er geen twee pagina's voor de huidige pagina zijn worden er
173      * meer nummers erna getoond. En vice versa. Het totaal van deze rij blijft
174      * altijd 5.</li>
175      * <li>Indien in de twee pagina's voor of na de huidige pagina (of in de
176      * uitzonderingsregel hierboven) de eerste of laatste pagina is opgenomen
177      * vervalt deze als aparte optie.</li>
178      * <li>Tussen de rij van 5 paginanummers en de eerste danwel laatste is een
179      * extra item met '...'.</li>
180      * <li>Afgezien van 'Vorige', 'Volgende' en '...' worden er alleen numerieke
181      * waarden getoond (de paginanummers)</li>
182      * </ul>
183      * </dd>
184      * </dl>
185      * 
186      * @param items
187      *            aantal items
188      * @param request
189      *            the request
190      * @return een html snippet met hyperlinks of een lege string als er geen
191      *         paginering is
192      */
193     protected final String buildPageList(int items, HttpServletRequest request) {
194         final int totaalAantalPaginas = (int) Math.ceil((double) items
195                 / this.itemsPerPage);
196         LOGGER.debug("Er komen " + totaalAantalPaginas + " pagina's");
197         if (totaalAantalPaginas <= 1) {
198             // bij 1 (of minder) geen paginering, klaar!
199             return "";
200         }
201 
202         int currentOffset;
203         try {
204             currentOffset = Integer.parseInt(request
205                     .getParameter(REQ_PARAM_PAGEOFFSET.code));
206         } catch (final NullPointerException e) {
207             currentOffset = 0;
208         } catch (final NumberFormatException e) {
209             currentOffset = 0;
210         }
211         final String reqString = this.buildQueryString(request,
212                 REQ_PARAM_PAGEOFFSET.code);
213 
214         final String addParam = request.getServletPath().replaceFirst("/", "")
215                 + "?" + reqString + "&amp;" + REQ_PARAM_PAGEOFFSET + "=";
216 
217         final int eerste = 0 * this.itemsPerPage;
218         final int vorige = currentOffset - this.itemsPerPage;
219         final int volgende = currentOffset + this.itemsPerPage;
220         final int laatste = (totaalAantalPaginas - 1) * this.itemsPerPage;
221 
222         final StringBuilder sb = new StringBuilder(
223                 "<ul class=\"geozetPaging\">");
224         // vorige
225         if (vorige >= 0) {
226             sb.append("<li class=\"prev first\"><a href=\"").append(addParam)
227                     .append(vorige).append("\">")
228                     .append(this._RESOURCES.getString("KEY_PAGINERING_VORIGE"))
229                     .append("</a></li>");
230         }
231         // eerste pag
232         if (eerste != currentOffset) {
233             sb.append(this.maakPaginaItem(addParam, eerste));
234         }
235 
236         LOGGER.debug("eerste=" + eerste + " huidige=" + currentOffset);
237         // ... tussendoor
238         if (totaalAantalPaginas > 6) {
239             // alleen als aantal pagina's groter is dan 5
240             if (currentOffset - eerste > 3 * this.itemsPerPage) {
241                 // alleen als er meer dan 3 paginas tussen eerste en huidige
242                 // zitten
243                 sb.append("<li>...</li>");
244             }
245         }
246 
247         // bepalen welke pagina's voorafgaand en volgend op
248         int start = currentOffset - (2 * this.itemsPerPage);
249         int maxPages = 5;
250         if (start < 0) {
251             // minimaal bij offset 0 beginnen
252             start = 0;
253         }
254 
255         if (currentOffset >= (totaalAantalPaginas - 3) * this.itemsPerPage) {
256             // maximaal bij totaal aantal minus 5 eindigen
257             start = (totaalAantalPaginas * this.itemsPerPage)
258                     - (maxPages * this.itemsPerPage);
259             if (start < 0) {
260                 // voor het geval er minder dan 5 pagina's zijn
261                 start = 0;
262                 maxPages = totaalAantalPaginas;
263             }
264 
265         }
266         // pagina's
267         int offset = 0;
268         for (int i = 0; i < maxPages; i++) {
269             offset = start + i * this.itemsPerPage;
270             if (currentOffset == offset) {
271                 LOGGER.debug("pagina " + (i + 1)
272                         + " is de huidige pagina en heeft offset " + offset);
273                 sb.append("<li class=\"active\"><strong>")
274                         .append(offset / this.itemsPerPage + 1)
275                         .append("</strong></li>");
276             } else {
277                 if (!((eerste == offset) || (laatste == offset))) {
278                     // eerste en laatste overslaan, die komen er sowieso in
279                     LOGGER.debug("pagina " + (i + 1) + " heeft offset "
280                             + offset);
281                     sb.append(this.maakPaginaItem(addParam, offset));
282                 }
283             }
284         }
285 
286         // ... tussendoor
287         if (totaalAantalPaginas > 6) {
288             if (laatste - currentOffset > 3 * this.itemsPerPage) {
289                 // alleen als er meer dan 3 paginas tussen laatste en huidige
290                 // zitten
291                 sb.append("<li>...</li>");
292             }
293         }
294 
295         // laatste pag
296         if (laatste != currentOffset) {
297             sb.append(this.maakPaginaItem(addParam, laatste));
298         }
299         // volgende alleen als we nog niet aan het einde zijn
300         if (volgende <= laatste) {
301             sb.append("<li class=\"next\"><a href=\"")
302                     .append(addParam)
303                     .append(volgende)
304                     .append("\">")
305                     .append(this._RESOURCES
306                             .getString("KEY_PAGINERING_VOLGENDE"))
307                     .append("</a></li>");
308         }
309         sb.append("</ul>");
310         return sb.toString();
311     }

313     /**
314      * genereer een pagina link.
315      * 
316      * @param addParam
317      *            request string snipper
318      * @param plusPag
319      *            pagina offset snipper
320      * @return pagina hyperlink
321      */
322     private String maakPaginaItem(String addParam, int plusPag) {
323         return "<li><a href=\"" + addParam + plusPag + "\">"
324                 + (plusPag / this.itemsPerPage + 1) + "</a></li>";
325     }
